find

  • find<T>(array: T, pred: IPredicate<T>, fromIndex?: number, wrap?: boolean): T
    • Defined in src/collections/algorithm.ts:303

    Find the first value in an array which matches a predicate.

    Type parameters

    • T

    Parameters

    • array: T

      The array of values to be searched.

    • pred: IPredicate<T>

      The predicate function to apply to the values.

    • Optional fromIndex: number

      The starting index of the search. If this value is negative, it is taken as an offset from the end of the array. If the adjusted value is still negative, it is clamped to 0.

    • Optional wrap: boolean

      Whether the search wraps around at the end of the array. If true and the end of the array is reached without finding the value, the search will wrap to the front of the array and continue until fromIndex - 1.

    Returns T

    The matching value or undefined if there is no match.

    Notes

    The range of visited indices is set before the first invocation of pred. It is not safe for pred to change the length of array.

    Example

    import algo = phosphor.collections.algorithm;

function isEven(value: number): boolean {
  return value % 2 === 0;
}

var data = [1, 2, 3, 4, 3, 2, 1];
algo.find(data, isEven);           // 2
algo.find(data, isEven, 4);        // 2
algo.find(data, isEven, -5);       // 4
algo.find(data, isEven, 6);        // undefined
algo.find(data, isEven, 6, true);  // 2

    See also findLast, findIndex, and indexOf.

indexOf

  • indexOf<T>(array: T, value: T, fromIndex?: number, wrap?: boolean): number
    • Defined in src/collections/algorithm.ts:42

    Find the index of the first occurrence of a value in an array.

    Type parameters

    • T

    Parameters

    • array: T

      The array of values to be searched.

    • value: T

      The value to locate in the array.

    • Default value fromIndex: number = 0

      The starting index of the search. If this value is negative, it is taken as an offset from the end of the array. If the adjusted value is still negative, it is clamped to 0.

    • Default value wrap: boolean = false

      Whether the search wraps around at the end of the array. If true and the end of the array is reached without finding the value, the search will wrap to the front of the array and continue until fromIndex - 1.

    Returns number

    The index of the value or -1 if the value is not found.

    Notes

    Values are compared using the strict equality === operator.

    Example

    import algo = phosphor.collections.algorithm;

var data = ['zero', 'one', 'two', 'three', 'two', 'one', 'zero'];
algo.indexOf(data, 'two');           // 2
algo.indexOf(data, 'two', 3);        // 4
algo.indexOf(data, 'two', -4);       // 4
algo.indexOf(data, 'two', 5);        // -1
algo.indexOf(data, 'two', 5, true);  // 2

    See also lastIndexOf, findIndex, and find.
