binaryFind

  • binaryFind<T, U>(array: T[], value: U, cmp: IComparator<T, U>): T
    • Defined in collections/algorithm.ts:606

    Using a binary search, find the first element in an array which compares == to a value.

    Type parameters

    • T

    • U

    Parameters

    • array: T[]

      The array of values to be searched. It must be sorted in ascending order.

    • value: U

      The value to locate in the array.

    • cmp: IComparator<T, U>

      The comparator function to apply to the values.

    Returns T

    The first element in array which compares == to value, or undefined if there is no such element.

    Example

    import algo = phosphor.collections.algorithm;

interface IPair {
  rank: number;
  value: string;
}

var data: IPair[] = [
  { rank: 0, value: 'zero' },
  { rank: 3, value: 'three' },
  { rank: 7, value: 'seven-A' },
  { rank: 7, value: 'seven-B' },
  { rank: 9, value: 'nine' },
];

function rankCmp(pair: IPair, rank: number): number {
  return pair.rank - rank;
}

algo.binaryFind(data, 7, rankCmp);  // { rank: 7, value: 'seven-A' }
algo.binaryFind(data, 8, rankCmp);  // undefined

    See also binaryFindLast and find.

find

  • find<T>(array: T[], pred: IPredicate<T>, fromIndex?: number, wrap?: boolean): T
    • Defined in collections/algorithm.ts:322

    Find the first value in an array which matches a predicate.

    Type parameters

    • T

    Parameters

    • array: T[]

      The array of values to be searched.

    • pred: IPredicate<T>

      The predicate function to apply to the values.

    • Optional fromIndex: number

      The starting index of the search. If this value is negative, it is taken as an offset from the end of the array. If the adjusted value is still negative, it is clamped to 0.

    • Optional wrap: boolean

      Whether the search wraps around at the end of the array. If true and the end of the array is reached without finding the value, the search will wrap to the front of the array and continue until one before fromIndex.

    Returns T

    The first matching value, or undefined if no value matches the predicate.

    Notes

    The range of visited indices is set before the first invocation of pred. It is not safe for pred to change the length of array.

    Example

    import algo = phosphor.collections.algorithm;

function isEven(value: number): boolean {
  return value % 2 === 0;
}

var data = [1, 2, 3, 4, 3, 2, 1];
algo.find(data, isEven);           // 2
algo.find(data, isEven, 4);        // 2
algo.find(data, isEven, -5);       // 4
algo.find(data, isEven, 6);        // undefined
algo.find(data, isEven, 6, true);  // 2

    See also findLast and binaryFind.
